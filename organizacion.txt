src/app/
│
├── core/                # Módulo Core: Solo importado en AppModule. Cosas de uso único global.
│   ├── guards/          # Guards de rutas (AuthGuard, RoleGuard)
│   ├── interceptors/    # Interceptores HTTP (TokenInterceptor, ErrorInterceptor)
│   ├── models/          # Interfaces y Clases modelo de datos de uso global
│   └── services/        # Servicios singleton globales (AuthService, ConfigService)
│                        # Nota: Si exam.service o facade.service son de uso global, podrían ir aquí.
│
├── shared/              # Módulo Shared: Importado en múltiples módulos funcionales.
│   ├── components/      # Componentes reutilizables "tontos"/presentacionales (Botones, Modales, Tablas)
│   ├── directives/      # Directivas personalizadas de uso común (ej. drag-and-drop, validaciones)
│   ├── pipes/           # Pipes reutilizables (ej. formateo de moneda especial, fechas)
│   └── utils/           # Funciones de utilidad sin dependencias de Angular
│
├── modules/ (o features/ o pages/)   # Agrupación por funcionalidades de negocio (Lazy Loading por defecto)
│   ├── auth/            # Módulo de Autenticación
│   │   ├── components/  # Componentes propios (LoginForm, RegisterForm)
│   │   ├── pages/       # Vistas enrutables (LoginPage, RegisterPage) -> equivalente a tus "screens"
│   │   └── services/    # Servicios específicos de autenticación
│   │
│   ├── exams/           # Módulo de Exámenes (Tu dominio de evaluar)
│   │   ├── components/  # Componentes propios de exámenes (ExamCard, QuestionList)
│   │   ├── pages/       # Vistas ruteables (ExamDashboard, TakeExam)
│   │   ├── models/      # Interfaces propias de exámenes
│   │   └── services/    # exam.service.ts y facade.service.ts podrían vivir aquí si solo se usan en exams.
│   │
│   └── dashboard/       # Otro módulo funcional...
│
├── layout/ (o partials/)# Layouts base de la aplicación
│   ├── header/          # Componentes del layout (Navbar)
│   ├── sidebar/         # Menú lateral
│   ├── footer/          # Pie de página
│   └── main-layout/     # Layout contenedor que agrupa header + sidebar + router-outlet
│
└── app.*                # routing, componentes raíz, etc. (app.component.ts, app.routes.ts)
